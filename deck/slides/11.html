SELECT p.id, p.content, p.like_count, p.comment_count, p.tip_total, p.created_at, u.username AS author_username, (l.user_id IS NOT NULL) AS liked_by_me FROM posts p JOIN users u ON u.id = p.author_id LEFT JOIN likes l ON l.post_id = p.id AND l.user_id = $viewer ORDER BY p.created_at DESC LIMIT 50;
