// app/actions/tips.ts — 트랜잭션 본문 await withTx(async (tx) => { // 1) 두 사용자 행 잠금 (id 오름차순) await tx.query("… WHERE id IN ($1,$2) FOR UPDATE"); // 2) 잔액 검증 — 부족하면 throw → ROLLBACK if (balance < amount) throw new Error("잔액 부족"); // 3) 잔액 이동 + 게시글 누적 await tx.query("UPDATE users SET credit_balance -= $1 …"); await tx.query("UPDATE users SET credit_balance += $1 …"); await tx.query("UPDATE posts SET tip_total += $1 …"); // 4) append-only 원장에 분개 기록 await tx.query("INSERT INTO credit_ledger (…) …"); });
